# Content-Similarity Layer Matching — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the Phase 1 size-closest baseline picker in `pkg/exporter.Planner` with byte-weighted tar-entry digest intersection, preserving the existing `min(patch, full)` guard and falling back to size-closest on any fingerprinting failure.

**Architecture:** A new `Fingerprint` type (`map[digest.Digest]int64`) and `Fingerprinter` interface with a `DefaultFingerprinter` impl land in `pkg/exporter/fingerprint.go`. The planner gains a `fingerprint` dependency and `baselineFP` cache populated lazily on first `Run` via `sync.Once`. `pickSimilar` walks a six-branch decision tree (target-fp-nil → size-fallback, all-candidate-fps-nil → size-fallback, max-score-zero → size-fallback, single winner, tie-broken by size, tie-broken by digest order). No sidecar change, no archive change, no CLI flag.

**Tech Stack:** Go 1.25, `github.com/klauspost/compress/zstd` (already a direct dep), `compress/gzip` + `archive/tar` (stdlib), `crypto/sha256`, `github.com/opencontainers/go-digest`, `testify/require`.

**Source spec:** `docs/superpowers/specs/2026-04-20-diffah-v2-content-similarity-matching-design.md` — §4 (architecture), §5 (flow), §7 (errors), §8 (testing), §9 (measurement gate). Read before executing.

//go:build ignore

// Command contentsim-spike compares size-closest vs byte-weighted
// tar-entry content-match layer picking on the Phase 1 POC blob pairs.
// Throwaway — delete after Task 0 produces a decision record.
//
// Usage:
//   go run scripts/contentsim-spike/main.go \
//       /tmp/diffah-poc/baselines.csv /tmp/diffah-poc/target.blob
//
// baselines.csv has lines of the form "<digest>,<path>".
package main

import (
	"archive/tar"
	"bytes"
	"compress/gzip"
	"crypto/sha256"
	"encoding/csv"
	"fmt"
	"io"
	"os"
	"os/exec"
	"path/filepath"
	"strings"
	"time"

	"github.com/klauspost/compress/zstd"
	"github.com/opencontainers/go-digest"
)

type baseline struct {
	digest digest.Digest
	path   string
	size   int64
	fp     fingerprint
	blob   []byte
}

type fingerprint map[digest.Digest]int64

func main() {
	if len(os.Args) != 3 {
		fmt.Fprintln(os.Stderr, "usage: contentsim-spike BASELINES.csv TARGET.blob")
		os.Exit(2)
	}

	baselines := mustLoadBaselines(os.Args[1])
	target, err := os.ReadFile(os.Args[2])
	must(err)
	targetFP := mustFingerprint(target, guessMediaType(os.Args[2]))
	targetSize := int64(len(target))

	// Pick size-closest.
	sizeWinner := closestBySize(baselines, targetSize)
	sizePatch := encodePatch(sizeWinner.path, os.Args[2])

	// Pick content-match (score each).
	startCM := time.Now()
	var best *baseline
	var bestScore int64
	for i := range baselines {
		s := score(targetFP, baselines[i].fp)
		if s > bestScore {
			bestScore = s
			best = &baselines[i]
		}
	}
	cmWall := time.Since(startCM)
	if best == nil {
		best = sizeWinner
	}
	contentPatch := encodePatch(best.path, os.Args[2])

	fmt.Printf("target            : %s (%d MB)\n", os.Args[2], targetSize/(1<<20))
	fmt.Printf("size-closest      : %s (%d MB) → patch %d B\n",
		filepath.Base(sizeWinner.path), sizeWinner.size/(1<<20), len(sizePatch))
	fmt.Printf("content-match     : %s (%d MB) → patch %d B (score %d)\n",
		filepath.Base(best.path), best.size/(1<<20), len(contentPatch), bestScore)
	fmt.Printf("content-match Δ   : %+d B (%.1f%%)\n",
		len(contentPatch)-len(sizePatch),
		100*float64(len(contentPatch)-len(sizePatch))/float64(len(sizePatch)))
	fmt.Printf("content-match wall: %s\n", cmWall)
}

func mustLoadBaselines(csvPath string) []baseline {
	f, err := os.Open(csvPath)
	must(err)
	defer f.Close()
	rows, err := csv.NewReader(f).ReadAll()
	must(err)
	out := make([]baseline, 0, len(rows))
	for _, r := range rows {
		blob, err := os.ReadFile(r[1])
		must(err)
		b := baseline{
			digest: digest.Digest(r[0]),
			path:   r[1],
			size:   int64(len(blob)),
			blob:   blob,
			fp:     mustFingerprint(blob, guessMediaType(r[1])),
		}
		out = append(out, b)
	}
	return out
}

func guessMediaType(path string) string {
	switch {
	case strings.HasSuffix(path, ".gz"):
		return "+gzip"
	case strings.HasSuffix(path, ".zst"):
		return "+zstd"
	default:
		return ""
	}
}

func mustFingerprint(blob []byte, mt string) fingerprint {
	var r io.Reader
	switch mt {
	case "+gzip":
		gz, err := gzip.NewReader(bytes.NewReader(blob))
		must(err)
		defer gz.Close()
		r = gz
	case "+zstd":
		zr, err := zstd.NewReader(bytes.NewReader(blob))
		must(err)
		defer zr.Close()
		r = zr
	default:
		r = bytes.NewReader(blob)
	}
	fp := make(fingerprint)
	tr := tar.NewReader(r)
	for {
		hdr, err := tr.Next()
		if err == io.EOF {
			return fp
		}
		must(err)
		if hdr.Typeflag != tar.TypeReg {
			continue
		}
		h := sha256.New()
		_, err = io.Copy(h, tr)
		must(err)
		d := digest.NewDigest(digest.SHA256, h)
		if _, exists := fp[d]; !exists {
			fp[d] = hdr.Size
		}
	}
}

func score(a, b fingerprint) int64 {
	var bytes int64
	for d, sz := range a {
		if _, ok := b[d]; ok {
			bytes += sz
		}
	}
	return bytes
}

func closestBySize(bs []baseline, want int64) *baseline {
	best := &bs[0]
	bestDelta := abs(best.size - want)
	for i := range bs[1:] {
		d := abs(bs[i+1].size - want)
		if d < bestDelta {
			best = &bs[i+1]
			bestDelta = d
		}
	}
	return best
}

func abs(x int64) int64 {
	if x < 0 {
		return -x
	}
	return x
}

func encodePatch(refPath, targetPath string) []byte {
	out, err := exec.Command("zstd", "-3", "--long=27", "--patch-from="+refPath, targetPath, "-c").Output()
	must(err)
	return out
}

func must(err error) {
	if err != nil {
		fmt.Fprintln(os.Stderr, err)
		os.Exit(1)
	}
}
```

//
// This file defines the byte-weighted fingerprint used by Planner to
// pick the most-content-similar baseline layer for each shipped layer.
// See docs/superpowers/specs/2026-04-20-diffah-v2-content-similarity-matching-design.md.

import "github.com/opencontainers/go-digest"

// Fingerprint of a decompressed tar layer: for each distinct regular-file
// content digest, the size of one instance. Directories, symlinks, hard
// links, and special files are skipped — they contribute no real bytes
// to zstd's patch-from window.
type Fingerprint map[digest.Digest]int64

// score returns the total byte-weighted intersection between target and
// candidate fingerprints. Nil inputs return 0.
func score(target, candidate Fingerprint) int64 {
	if target == nil || candidate == nil {
		return 0
	}
	var bytes int64
	for d, size := range target {
		if _, ok := candidate[d]; ok {
			bytes += size
		}
	}
	return bytes
}

// Package exporter — content-similarity helpers.
//
// See docs/superpowers/specs/2026-04-20-diffah-v2-content-similarity-matching-design.md.
package exporter

import (
	"archive/tar"
	"bytes"
	"context"
	"crypto/sha256"
	"errors"
	"fmt"
	"io"

	"github.com/opencontainers/go-digest"
)

// Fingerprint of a decompressed tar layer: for each distinct regular-file
// content digest, the size of one instance. Directories, symlinks, hard
// links, and special files are skipped — they contribute no real bytes
// to zstd's patch-from window.
type Fingerprint map[digest.Digest]int64

// score returns the total byte-weighted intersection between target and
// candidate fingerprints. Nil inputs return 0.
func score(target, candidate Fingerprint) int64 {
	if target == nil || candidate == nil {
		return 0
	}
	var bytes int64
	for d, size := range target {
		if _, ok := candidate[d]; ok {
			bytes += size
		}
	}
	return bytes
}

// Fingerprinter hashes a compressed layer blob into a Fingerprint.
// Media type picks the decompressor; unknown / malformed input yields
// an error wrapping ErrFingerprintFailed.
type Fingerprinter interface {
	Fingerprint(ctx context.Context, mediaType string, blob []byte) (Fingerprint, error)
}

// ErrFingerprintFailed is the sentinel wrapped by every error returned
// from DefaultFingerprinter.Fingerprint. Planner branches use
// errors.Is(err, ErrFingerprintFailed) to fall back to size-closest.
var ErrFingerprintFailed = errors.New("fingerprint failed")

// DefaultFingerprinter handles plain tar (initially). gzip and zstd
// compressors are added in later tasks.
type DefaultFingerprinter struct{}

// Fingerprint implements Fingerprinter. This first revision handles only
// plain tar; it always opens the blob as a tar reader. Subsequent tasks
// extend to gzip+tar and zstd+tar layers.
func (DefaultFingerprinter) Fingerprint(
	ctx context.Context, mediaType string, blob []byte,
) (Fingerprint, error) {
	return fingerprintTar(ctx, bytes.NewReader(blob))
}

// fingerprintTar streams through a tar reader, hashing every regular-file
// entry's content and deduplicating by digest.
func fingerprintTar(ctx context.Context, r io.Reader) (Fingerprint, error) {
	tr := tar.NewReader(r)
	fp := make(Fingerprint)
	for {
		if err := ctx.Err(); err != nil {
			return nil, fmt.Errorf("%w: %w", ErrFingerprintFailed, err)
		}
		hdr, err := tr.Next()
		if errors.Is(err, io.EOF) {
			break
		}
		if err != nil {
			return nil, fmt.Errorf("%w: tar header: %w", ErrFingerprintFailed, err)
		}
		if hdr.Typeflag != tar.TypeReg {
			continue
		}
		h := sha256.New()
		if _, err := io.Copy(h, tr); err != nil {
			return nil, fmt.Errorf("%w: tar body: %w", ErrFingerprintFailed, err)
		}
		d := digest.NewDigest(digest.SHA256, h)
		if _, exists := fp[d]; !exists {
			fp[d] = hdr.Size
		}
	}
	return fp, nil
}
```

func (p *Planner) ensureBaselineFP(ctx context.Context) {

			if err != nil {
				p.baselineFP[b.Digest] = nil
				continue
			}
			p.baselineFP[b.Digest] = f

Add to `pkg/exporter/intralayer.go` below `pickClosest`:

```go
// pickSimilar chooses the baseline most content-similar to the target
// (byte-weighted tar-entry digest intersection), falling back to
// pickClosest on any of three unrecoverable conditions:
//
//   1. targetFP is nil (target fingerprinting failed)
//   2. no baseline has a non-nil fingerprint
//   3. every candidate's score is 0 (no shared content)
//
// Ties on score break by size-closest; further ties break by the
// baseline's index in the sorted-by-digest p.baseline slice.
func (p *Planner) pickSimilar(
	targetFP Fingerprint, targetSize int64,
) (BaselineLayerMeta, bool) {
	if len(p.baseline) == 0 {
		return BaselineLayerMeta{}, false
	}
	if targetFP == nil {
		return p.pickClosest(targetSize)
	}

	// Collect candidates that actually have fingerprints and score them.
	type scored struct {
		meta  BaselineLayerMeta
		score int64
	}
	cands := make([]scored, 0, len(p.baseline))
	for _, b := range p.baseline {
		fp := p.baselineFP[b.Digest]
		if fp == nil {
			continue
		}
		cands = append(cands, scored{meta: b, score: score(targetFP, fp)})
	}
	if len(cands) == 0 {
		return p.pickClosest(targetSize)
	}

	// Determine max score.
	var maxScore int64
	for _, c := range cands {
		if c.score > maxScore {
			maxScore = c.score
		}
	}
	if maxScore == 0 {
		return p.pickClosest(targetSize)
	}

	// Narrow to winners.
	winners := cands[:0]
	for _, c := range cands {
		if c.score == maxScore {
			winners = append(winners, c)
		}
	}
	if len(winners) == 1 {
		return winners[0].meta, true
	}

	// Tie-break on size-closest, then first in sorted-by-digest order.
	best := winners[0].meta
	bestDelta := absDelta(best.Size, targetSize)
	for _, w := range winners[1:] {
		d := absDelta(w.meta.Size, targetSize)
		if d < bestDelta {
			best, bestDelta = w.meta, d
		}
	}
	return best, true
}
```

// buildSharedLayerBlob creates a gzipped tar containing multiple named
// files, returning (compressed bytes, diffID, digest). Every file is a
// regular-file entry so they count in a Fingerprint.
func buildSharedLayerBlob(files []layerFile) ([]byte, digest.Digest, digest.Digest) {
	rawBuf := &bytes.Buffer{}
	tw := tar.NewWriter(rawBuf)
	for _, f := range files {
		hdr := &tar.Header{
			Name:     f.name,
			Size:     int64(len(f.data)),
			Mode:     0o644,
			ModTime:  pinnedTime,
			Typeflag: tar.TypeReg,
			Uid:      0, Gid: 0, Uname: "", Gname: "",
		}
		if err := tw.WriteHeader(hdr); err != nil { panic(err) }
		if _, err := tw.Write(f.data); err != nil { panic(err) }
	}
	if err := tw.Close(); err != nil { panic(err) }
	rawBytes := rawBuf.Bytes()
	diffID := digest.FromBytes(rawBytes)

	gzBuf := &bytes.Buffer{}
	gw, _ := gzip.NewWriterLevel(gzBuf, gzip.BestCompression)
	gw.ModTime = pinnedTime
	if _, err := gw.Write(rawBytes); err != nil { panic(err) }
	if err := gw.Close(); err != nil { panic(err) }
	gzBytes := gzBuf.Bytes()
	return gzBytes, diffID, digest.FromBytes(gzBytes)
}

type layerFile struct {
	name string
	data []byte
}

// overlapFiles returns `shared` "file-i" entries with content-digest
// that matches between baseline and target, plus `unique` entries that
// are unique to each side (differing seed).
func overlapFiles(sharedCount, uniqueCount, shareSeed, uniqueSeed int, sharedSize, uniqueSize int) []layerFile {
	out := make([]layerFile, 0, sharedCount+uniqueCount)
	for i := 0; i < sharedCount; i++ {
		out = append(out, layerFile{
			name: fmt.Sprintf("shared/%04d", i),
			data: pseudoRandomBytes(int64(shareSeed+i), sharedSize),
		})
	}
	for i := 0; i < uniqueCount; i++ {
		out = append(out, layerFile{
			name: fmt.Sprintf("unique/%04d", i),
			data: pseudoRandomBytes(int64(uniqueSeed+i), uniqueSize),
		})
	}
	return out
}

// pseudoRandomBytes is deterministic across runs given a seed.
func pseudoRandomBytes(seed int64, n int) []byte {
	src := rand.NewSource(seed)
	r := rand.New(src)
	b := make([]byte, n)
	_, _ = r.Read(b)
	return b
}
